class ContactInfo { 
    private String myTitle; 
    private String myContent; 
 
    protected ContactInfo(String title,String content){ 
        myContent = content; 
        myTitle = title; 
    } 
 
    protected String getMyTitle(){ 
        return myTitle; 
    } 
 
    protected String getMyContent(){ 
        return myContent; 
    } 
 
    public String toString(){ 
        return myTitle + " " + myContent; 
    } 
 
}
